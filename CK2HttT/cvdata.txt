country_conversion = {

#	We assume that the tech group is defined by a combination of culture (primary_culture only) and
#		religion.
#	The fields you put here are the rules to use to choose the appropriate tech group.
#	The name of the field is the tech group the rule applies to (should be western, eastern or
#		muslim; slim chance for any other)
#	Field content is:
#		religion		= religion_name
#		religious_group	= religion_group_name
#		culture		= culture_name
#		culture_group	= culture_group_name
#		prio			= a number
#	There should be at least one of the four first fields, and at most only one for each subgroups
#		(religion,religion_group), (culture,culture_group).
#	The prio fields indicates the priority of the rule: the most general rules should have a low
#		priority (0) while the most specific rules should take precedence with a high number.
#	The lowest priority rules should be able to cover all situations. 

	tech_group = {
		western = { prio=0 religious_group=christian }
		muslim  = { prio=0 religious_group=muslim }
		indian  = { prio=0 religious_group=pagan }
		eastern = { prio=1 culture_group=south_slavic }
		eastern = { prio=1 culture_group=west_slavic }
		eastern = { prio=1 culture_group=east_slavic }
		eastern = { prio=1 culture_group=byzantine }
		eastern = { prio=2 culture=turkish }
		western = { prio=2 culture=slovenian }
		western = { prio=2 culture=czech }
		western = { prio=2 culture=schlesian }
		eastern = { prio=2 culture=berber }
		eastern = { prio=2 culture=maltese }
		eastern = { prio=2 culture=maltese }
	}

#	SLIDERS: note that although the described rules may differ widely, they all use the same
#		format, so that anything you see in one can be used in another (for example, you could use
#		buildings in the mercantilism_freetrade rule or laws in the land_naval one.
#		In the same way, all buildings can be used in the buildings sub-rule, and all advances in
#		the advances sub-rule.

#	Government type is chosen along a similar scheme as tech_group.
#	There is one additional optional parameter, which is the government form under CK, and another
#		that are laws used by the realm (more than one allowed)
	government = {
		HttT = {
			use = { type=stand_alone }
			convert = {
				tribal_federation	= { prio=0 religious_group=pagan }
				theocratic_government	= { prio=0 form=ecclestical }
				theocratic_government	= { prio=0 form=knight_order }
				merchant_republic	= { prio=0 form=republican }
				feudal_monarchy		= { prio=0 form=feudal religious_group=christian }
				despotic_monarchy	= { prio=1 form=feudal law=royal_preorgatory_law religious_group=christian }
				administrative_republic	= { prio=1 form=republican religious_group=christian culture_group=germanic }
				despotic_monarchy	= { prio=0 religious_group=muslim }
				tribal_despotism	= { prio=1 religious_group=muslim culture=bedouin_arabic }
				tribal_despotism	= { prio=1 religious_group=muslim culture=al_iraqiya_arabic }
				tribal_despotism	= { prio=1 religious_group=muslim culture=al_suryah_arabic }
				tribal_federation	= { prio=1 religious_group=muslim culture_group=persian }
			}
		}
	}

#	Land-Naval policy is set by comparing the number of counties in CK to the modified number of
#		equipment ports (wharf, harbors, etc...); the total will be averaged and compared to the
#		step: each step is worth 1 point.
	land_naval = {
		buildings = {
			fishing_wharf		= 1
			civilian_harbour	= 3
			naval_harbour		= 7
			grand_shipyard		= 11
		}
		zero = -5
		step =  2
	}

#	Mercantilism-Free Trade is set by a similar way, but counting the advances in Commerce
#	Institutions, Trade Practices, Fairs and Routes. It is assumed that the more advanced you are,
#	the more mercantilist you would be (mercantilism would be perceived high-tech at the time).
	mercantilism_freetrade = {
		advances = {
			minor_merchant_houses		= 0
			local_merchant_houses		=-1

			international_fairs		=-8
		}
		zero =  5
		step =  3
	}

#	Serfdom-Freesubject is set by averaging the counties peasant power over the country.
#	Laws may also influence the final result.
	serfdom_freesubjects = {
		power = {
			peasants =1.0
		}
		zero = -5
		step =  0.05
		laws = {
			royal_preorgatory_law	=-2
			feudal_contract_law	=-1
			traditional_custom_law	= 0
			popular_law		= 1
		}
	}

#	Centralization-Decentralization is set by computing an average on all counties, depending on its
#		level of rule: direct (king's own demesne), count (under the rule of a count under the
#		direct rule of the king), other (under the rule of any other character). Only the
#		counties actually inside the EU3 realm are counted : those finally belonging to
#		union/vassal/allied realms are not.
#	Laws may also influence the final result.
	centralization_decentralization = {
		hierarchy = {
			direct	=-4
			count	=-2
			other	= 4
		}
		zero = 5

#-------------------------------------------------
#rules for making a country when it is a CK vassal
#-------------------------------------------------
#	For most fields, you will put on the right a structure the four possible choices (you don't have
#			to fill up all of these; all are optional):
#		mod: (see final effect structure) a number between -1000 and 1000
#			A positive number increases the distance from the liege 
#			A negative number decreases the distance from the liege
#		best, worst:
#			merged, union, vassal, ally, marriage, rNNN (relationship value, 0 to 200)
#			For best, this state will be the best possible relationship
#			For worst, the worst possible; of course, you must have best>=worst
#			Use these sparingly: the result of large conflicts is not easy to guess.
#			If more than one of these conditions is met, see below. We will try to meet the
#				lowest best and highest worst.
#			NOTE: as of now, rNNN have NO effect 
#		prio: a number between -100 and 100
#			A number used to resolve conflicts between conflicting conditions for best/worst;
#			in short, a priority marker for selecting between conflicting situations.
#			In details: at the end of the analysis, you get a series of
#				worst (p1) worst (p2)... worst (pn) (decreasing order of p)
#				best  (p1) best  (p2)... best  (pn) (decreasing order of p)
#				if min(best) >= max(worst) there is no conflict.
#				Otherwise, we remove the best=min(best) or worst=max(worst) with the lowest
#				priority and iterate until there is no conflict.
#		tier:
#			county or duchy: you may differentiate rules for both tiers.
#			The rule applies when converting countries from that tier.
#			If not present, the rule applies to both tiers equally.
#
#	For example, weak_crown { GERM = { mod = 3 best = vassal worst = ally } }
#		would indicate that all german duchies would end up as either vassals or allies of their
#		lord (GERM), depending on the other conditions: they would never merge.
#	Also note that mod tables do have an inherent order assumed, and that breaking that basic
#		supposition may yield strange results. For exemple, it is assumed that the more loyal a
#		vassal is, the more he is likely to stick with his liege. If you put values that do not
#		follow this order, then things become unpredictable.

country_convert_rules = {
#	accepted culture:
#		For the purpose of determining counties which are "close" or "far" from their liege.
#		- liege if the ruler culture is accepted
#		- capital if the capital culture is accepted
#		- major_by_income if the dominant culture weighted with the province income is accepted
#		- major_by_province if the dominant culture by counting provinces is accepted
	accepted_cultures = { ruler capital major_by_income }

#	liege country modifier
#		Historically, some crowns were bad at keeping their vassals in their folds while others
#		were good.
	weak_crown = {
		GERM = { mod=1 best=vassal worst=r150 tier=duchy }
		ITAL = { mod=1 best=vassal worst=r100 tier=duchy }
	}

#	individual country modifier
#		This allows individual settings for tags.
#		It works as for weak_crown, but for specific countries; the result only applies to that
#			country in relation to its lord.
#		With this, you can decide that one specific country will have a greater chance to exist,
#			or even force it to be independent.
#		In the base version of the converter, this is not used; but you can use it for specific
#			needs.
	country_modifier = {  
		#C082 = { mod=2 best=vassal worst=r150 }
	}

#	The liege's piety, prestige or gold may influence the way his vassals see him ; in the reverse,
#		a vassal with too much gold/prestige or too low piety may think himself greater than he
#		should be.
#	This is taken into account by a complex rule:
#		L is the liege value, V the vassal's one:
#			* If L<0, then the minimal entry in the table is used
#			* If V<0, then the maximal entry in the table is used
#			* If both conditions apply, then both results are used
#			* if L*L+V*V <= radius*radius, then the result is 0
#				(too low values for both aren't significant)
#			* Otherwise, we check V/L as a percentage in the table
	prestige_radius = 200
	prestige = {
		100	= { mod= 1 }	#vassal prestige > liege prestige
		20	= { mod= 0 }
		0	= { mod=-1 }	#liege has at least 5*vassal prestige
	}
	gold_radius = 2000
	gold = {
		200	= { mod= 1 }	#vassal gold > 2*liege gold
		5	= { mod= 0 }
		0	= { mod=-1 }	#liege has at least 20*vassal gold
	}
#	For piety, it's a bit simpler because vassal would not try to
#		compare his score to his liege's score.
	liege_piety = {
		5000	= { mod=-1 }	#high piety: vassals tend to stick. RUC: increased to 5000 from 1500
		100	= { mod=0 }		#medium piety: no effect
		-9999	= { mod=1 }		#poor piety: vassal tend to distance themselves
	}

	vassal_piety = {
		500	= { mod=-1 }	#high piety: vassal tends to stick
		-9999	= { mod=0 }		#poor piety: no effect
	}

#	The liege personality may have influence
#	Possible fields are:
#		trait = value => to indicate positive/negative modifiers for the given trait
	liege_traits = {
		realm_duress = { mod=1 }
	}

#	The kind of crown may have influence
#		The fields are simple here! beware!
	tier = {
		#the normal ones are below (RUC)
		county=-2	# slightly less than Yves' normal
		duchy=0
		kingdom=2	#rare but can happen (jerusalem)

		#special massive-blobification results below.
		#county=-10
		#duchy=-10
		#kingdom=-10	 #rare but can happen (jerusalem)
	}

#	Vassal crowns: if the vassal has crowns on which the liege has no claim
#		only one level of crown is counted (i.e. this has no effects for county crowns when
#		comparing to a king!)
#	number of crowns = modifier (not cumulative)
	own_crowns = {
		#1 = { mod=1 }
		#2 = { mod=1 best=vassal  }
		#3 = { mod=2 best=vassal  }
	}

#	Liege doesn't hold the appropriate crown over the vassal's capital county
	liege_crowns = {
		not_king = { mod=2 }	#applies only if liege is king level
		not_duke = { mod=2 tier=county }
	}

#	Loyalty modifier
#	Possible fields are:
#		numbers from 0 to 100 in decreasing order: effect if the ruler loyalty is below (<=) the #		number (not cumulative)
	loyalty = {
		60 = { mod=1 }
		30 = { mod=2 }
		10 = { mod=3 }
	}

#	Rule specific to counties vassal to a duchy whose duke has decided to merge
	merged_liege = { mod=-2 tier=county }

#	The vassal personality may have influence
#	Possible fields are:
#		trait = value	=> to indicate positive/negative modifiers for the given trait
	traits = {
		rebellious	= { mod= 1 }
		loyalist	= { mod=-1 }
		just		= { mod=-1 }
	}

#	Different culture may have influence; to the difference of other indicators, culture is hard to
#		guess, because as long as the final country is unknown, it's precise culture is unknown
#		too. So the culture thing must be seen as a best guess as regards any number based on the
#		"majority".
#	Possible fields are:
#		ruler: effect if the ruler culture is not accepted
#		capital: effect if the capital culture is not accepted
#		numbers from 0 to 100 in increasing order: effect if that percentage of the duchy culture
#				(>=) is not accepted (not cumulative)
#			This percentage is computed depending on the accepted_cultures setting; if both
#			major_by_income and major_by_province are set, the modifier will be computed on the
#			base of the average of both ratios.
	culture = {
		ruler		= { mod=1 }
		capital		= { mod=1 tier=duchy }
		50		= { mod=1 }
		80		= { mod=2 }
	}

#	Global size (including own CK vassals if any -before conversion-) may have influence.
#	Possible fields are:
#		numbers, in increasing order: effect if the size is that many counties or more (>=)
#			(not cumulative).
	size = {
		0	= { mod=0 }		#hum... what does a duke with no demesne think ???
		1	= { mod=-1 }	#a small demesne is less likely to become independent
		2	= { mod=0 }
		4	= { mod=1 }
		7	= { mod=2 }
		10	= { mod=3 }
	}

#	After adding the modifiers together, check the results.
#	The constraints gained previously may modify that result.
	effect = {
		3	= union	#country becomes personal union
		5	= vassal	#country becomes vassal
		8	= ally	#country becomes ally
		11	= marry	#country becomes married
		15	= r150	#country becomes friend (100 relationship)
	}
}

# This section defines how to convert from CK tags to EU3 tags
# It contains one or more subsection, one for each possible set of conversion
# The name for a given set must be also defined in the "selection" section below or it won't be seen.
# Each subsection contains a number of mandatory and ioptional fields:
#   use =    mandatory, contains two subfields:
#         type = stand_alone or differential
#         base = name : in case the modification is slight, use the base definition and apply the differences
#                       defined in this section ; you can refer to another differential section, but should
#                       end up falling back on a stand_alone section.
#   convert =  optional, contains a list of simple subfields in the form:
#         cktag = eu3tag   or
#         cktag = "eu3tag1,eu3tag2,...eu3tagN"
#              these define the rules to convert a CK tag to an EU3 tag in preferential order (first is prefered)
#   tag_pool = optional (cannot exist empty), contains possibly two optional subfields (don't use empty ones!)
#         add = { TAG1 TAG2 ... TAGN }   add these tags to the tag pool
#         rem = { TAG1 TAG2 ... TAGN }   remove these tags from the tag pool
#         The tag pool is used to list the tags that may be chosen randomly if no other rule applies ; that's the
#         last resort chance to get a tag. This will always be done last, when all countries have had their chance
#         to get their tag.
#   unavailable = optional, exact same format as tag_pool
#         The list is used to render a tag unusable, whatever may have been written before
#         First, these tags are removed from any tag pool
#         Second, any tag in the convert section that refers to an unavailable tag is simply ignored
#         This section is mostly used for:
#         - removing some tags from selection (for exemple, you may not want Germany to come as a formed country)
#         - removing some tags from selection because you used a convert section that is broad and uses some tags
#           that are not allowed under the current option

#this is used to select which version of EU3 to install to
#it contains one first field:
#    autodetect=XXX where XXX is either
#               true : try to autodetect
#               false : don't try to autodetect (popup)
#               (version) : set the version to use (e.g. NA = {...})
#and a list of other fields in the form of:
#    XXXX = { name="YYYY" autodetect="ZZZZ" }
#        XXXX is the name of the version (what you want
#             but in sync with what is declared in the previous sections :
#             culture, province, tags
#        YYYY is the string that will be shown to the user if autodetection
#             is disabled or fails
#        ZZZZ is either the string "default" (default choice if all other fail)
#             or a comma separated list of country tags specific to that version
#the order is important because the last entry that matches is accepted.
#
#example:
#    NA =  { name = "Napoleon Ambition"    autodetect="DNZ,KRA,LIV,RIG" }
#
selection = {
    autodetect=true
    # autodetect=false
    HttT =  { name = "Heir to the Throne"            autodetect="default" }
